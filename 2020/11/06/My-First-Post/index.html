CREATE PROCEDURE productpricing(
    out pl DECIMAL(8,2),
    out ph DECIMAL(8,2),
    out pa DECIMAL(8,2)
)
BEGIN
    SELECT Min(prod_price)
    INTO pl
    FROM products;
    SELECT Max(prod_price)
    INTO ph
    FROM products;
    SELECT Avg(prod_price)
    INTO pa
    FROM products;
END //
DELIMITER ;
